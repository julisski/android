CloudSync shows cloud storage used together with Room — the offline-first pattern. enum class SyncState { SYNCED, PENDING, FAILED }    // the outbox marker

@Entity(tableName = "notes")
data class Note(
    @PrimaryKey val id: String,            // CLIENT-generated UUID (an offline row has an id already)
    val title: String, val body: String,
    val updatedAt: Long,                   // last-write-wins clock
    val syncState: SyncState = SyncState.PENDING,
    val deleted: Boolean = false,          // soft-delete tombstone (pushable while offline)
)

@Serializable

fun Note.toDto() = NoteDto(id, title, body, updatedAt, deleted)
fun NoteDto.toEntity(state: SyncState = SyncState.SYNCED) = Note(id, title, body, updatedAt, state, deleted)

interface NoteDao {
    @Query("SELECT * FROM notes WHERE deleted = 0 ORDER BY updatedAt DESC")
    fun observeVisibleNotes(): Flow<List<Note>>            // tombstones hidden

    @Query("SELECT COUNT(*) FROM notes WHERE syncState != 'SYNCED'")

data/CloudApi.kt — Real vs Fake + the switch
interface CloudApi {
    suspend fun pullSince(since: Long): List<NoteDto>
    suspend fun push(note: NoteDto): NoteDto
}

class FakeCloudApi : CloudApi {                            // offline, in-memory "server"
    private val server = linkedMapOf<String, NoteDto>()
    override suspend fun pullSince(since: Long): List<NoteDto> {
        delay(700); return server.values.filter { it.updatedAt > since }.sortedBy { it.updatedAt }
    }
    override suspend fun push(note: NoteDto): NoteDto {
        delay(500)
        val existing = server[note.id]
        if (existing == null || note.updatedAt >= existing.updatedAt) server[note.id] = note   // server LWW
        return server.getValue(note.id)
    }
}

data/NoteRepository.kt — optimistic writes + sync() + LWW
// PURE last-write-wins rule (unit-tested with no Room/coroutines).
fun shouldAcceptRemote(local: Note?, remote: NoteDto): Boolean =
    local == null || (local.syncState == SyncState.SYNCED && remote.updatedAt >= local.updatedAt)

suspend fun addNote(title: String, body: String) {
    dao.upsert(Note(UUID.randomUUID().toString(), title.trim(), body.trim(),
        System.currentTimeMillis(), SyncState.PENDING))    // optimistic, joins the outbox
}

suspend fun sync() {
    for (local in dao.notesToPush()) {                     // 1) PUSH the outbox
        val serverCopy = cloud.push(local.toDto())         // on failure: mark FAILED + rethrow → worker retries
        if (local.deleted) dao.hardDelete(local.id) else dao.upsert(serverCopy.toEntity(SyncState.SYNCED))
    }
    val since = dao.maxUpdatedAt() ?: 0L                   // 2) PULL remote changes
    for (remote in cloud.pullSince(since)) {
        if (shouldAcceptRemote(dao.findById(remote.id), remote))
            if (remote.deleted) dao.hardDelete(remote.id) else dao.upsert(remote.toEntity(SyncState.SYNCED))
    }
}

sync/SyncWorker.kt + SyncScheduler.kt — background, network-gated
class SyncWorker(appContext: Context, params: WorkerParameters) : CoroutineWorker(appContext, params) {
    override suspend fun doWork(): Result {
        val repo = NoteRepository.get(applicationContext)   // the SAME repo the UI uses
        return try { repo.sync(); Result.success() }
        catch (e: Exception) { Result.retry() }            // backoff; try again later
    }
}

// The constraint is WHY offline just queues:
OneTimeWorkRequestBuilder<SyncWorker>()
    .setConstraints(Constraints.Builder()
        .setRequiredNetworkType(NetworkType.CONNECTED)     // run online, DEFER offline
        .build())
    .setBackoffCriteria(BackoffPolicy.EXPONENTIAL, 30, TimeUnit.SECONDS)
    .build()

Key idea · who wins?

Conflict resolution — last-write-wins

Both devices edit the same note while offline, then both sync. Something must decide whose version survives. CloudSync compares updatedAt — the later edit wins. Make each device edit and watch the rule (shouldAcceptRemote) pick a winner.

Key idea · one-line offline switch

Real vs Fake — the same interface, two clouds
